```typescript
private detectProjectileHits() {
    for (const [projId, proj] of this.state.projectiles.entries()) {
        const owner = this.state.players.get(proj.ownerId);
        if (!owner) continue;
        
        for (const [targetId, target] of this.state.players.entries()) {
            if (targetId === proj.ownerId) continue;
            if (target.isDead) continue;
            
            const radius = this.getPlayerRadius(target);
            const dx = proj.x - target.x;
            const dy = proj.y - target.y;
            const distSq = dx * dx + dy * dy;
            
            if (distSq <= radius * radius) {
                // Hit detected
                this.applyProjectileDamage(owner, target, proj.damage);
                this.state.projectiles.delete(projId);
                break;
            }
        }
    }
}

private applyProjectileDamage(attacker: Player, victim: Player, damage: number) {
    // Check invulnerability
    if (this.tick < victim.invulnerableUntilTick) return;
    if (victim.isLastBreath) return;
    
    // Check shield
    if (this.tick < victim.shieldEndTick) {
        victim.shieldEndTick = 0; // Shield blocks, but is consumed
        attacker.lastAttackTick = this.tick;
        return;
    }
    
    // Apply damage (similar to processCombat)
    const classStats = this.getClassStats(attacker);
    const finalDamage = damage * classStats.damageMult;
    
    const massLoss = victim.mass * finalDamage;
    const attackerGain = massLoss * this.balance.combat.pvpBiteAttackerGainPct;
    const scatterMass = massLoss * this.balance.combat.pvpBiteScatterPct;
    
    victim.mass -= massLoss;
    this.applyMassDelta(attacker, attackerGain);
    this.spawnPvPBiteOrbs(victim, scatterMass);
    
    victim.invulnerableUntilTick = this.tick + this.invulnerableTicks;
    attacker.lastAttackTick = this.tick;
    
    // Check death
    if (victim.mass <= this.balance.physics.minSlimeMass) {
        this.handlePlayerDeath(victim, attacker);
    }
}
```
